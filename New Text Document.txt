        private Dictionary<int, string> GetJobHistory(DateTime targetDate)
        {
            var results = new Dictionary<int, string>();
            int sqlTargetDate = Convert.ToInt32(targetDate.ToString("yyyyMMdd"));

            string query = @"
            WITH NonBusinessHolidays2026 AS (
                SELECT '2026-04-02' AS HolidayDate UNION ALL SELECT '2026-04-08' UNION ALL SELECT '2026-04-09' UNION ALL
                SELECT '2026-04-22' UNION ALL SELECT '2026-05-21' UNION ALL SELECT '2026-05-22' UNION ALL
                SELECT '2026-09-11' UNION ALL SELECT '2026-09-12' UNION ALL SELECT '2026-09-13' UNION ALL
                SELECT '2026-09-20' UNION ALL SELECT '2026-09-21' UNION ALL SELECT '2026-09-25' UNION ALL
                SELECT '2026-09-26' UNION ALL SELECT '2026-10-02' UNION ALL SELECT '2026-10-03'
            ),
            TargetMonthDays AS (
                SELECT CAST(DATEADD(month, DATEDIFF(month, 0, GETDATE()), 0) AS DATE) AS CheckedDate
                UNION ALL
                SELECT DATEADD(DAY, 1, CheckedDate)
                FROM TargetMonthDays
                WHERE CheckedDate < CAST(GETDATE() AS DATE)
            ),
            BusinessDaysCount AS (
                SELECT COUNT(*) AS PassedBusinessDays
                FROM TargetMonthDays d
                LEFT JOIN NonBusinessHolidays2026 h ON d.CheckedDate = h.HolidayDate
                WHERE DATENAME(WEEKDAY, d.CheckedDate) NOT IN ('Friday', 'Saturday')
                  AND h.HolidayDate IS NULL
            ),
            JobList AS (
                SELECT 1 AS JobOrder, N'DailyProcess' AS JobName UNION ALL
                SELECT 2 AS JobOrder, N'Pension_Daily_Sync' AS JobName UNION ALL
                SELECT 4 AS JobOrder, N'Import External Data' AS JobName UNION ALL
                SELECT 9 AS JobOrder, N'Pension_HandleTransferFileAndCreateTransferOutPension' AS JobName
            ),
            JobHistory AS (
                SELECT job_id, run_status, run_date, run_time,
                ROW_NUMBER() OVER (PARTITION BY job_id ORDER BY run_date DESC, run_time DESC) AS rn
                FROM msdb.dbo.sysjobhistory WHERE step_id = 0
                AND run_date >= @MinDate
            ),
            LatestHistory AS (
                SELECT job_id, run_status, run_date, run_time FROM JobHistory WHERE rn = 1
            )
            SELECT 
                jl.JobOrder, 
                ISNULL(h.run_status, -1) as Status,
                (SELECT PassedBusinessDays FROM BusinessDaysCount) AS BusinessDays,
                CASE WHEN h.run_date IS NOT NULL THEN
                    CONVERT(DATETIME, STUFF(STUFF(RIGHT('00000000' + CAST(h.run_date AS VARCHAR(8)), 8), 5, 0, '-'), 8, 0, '-') + ' ' + 
                    CONVERT(VARCHAR(8), STUFF(STUFF(RIGHT('000000' + CAST(h.run_time AS VARCHAR(6)), 6), 3, 0, ':'), 6, 0, ':')))
                ELSE NULL END as RunDateTime
            FROM JobList jl
            INNER JOIN msdb.dbo.sysjobs j ON j.name = jl.JobName
            LEFT JOIN LatestHistory h ON j.job_id = h.job_id
            OPTION (MAXRECURSION 100);";

        private Dictionary<string, string> GetBIJobStatuses()
        {
            var biResults = new Dictionary<string, string>(StringComparer.OrdinalIgnoreCase);
            string query = @"
                SELECT 
                    j.name AS JobName,
                    CASE js.last_run_outcome
                        WHEN 1 THEN 'הסתיים בהצלחה'
                        WHEN 0 THEN '<b>נכשל או טרם הסתיים</b>'
                        ELSE '<b>UNKNOWN</b>'
                    END AS LastStatus
                FROM msdb.dbo.sysjobs j
                JOIN msdb.dbo.sysjobservers js ON j.job_id = js.job_id;";

            try
            {
                using (SqlConnection conn = new SqlConnection(connStrForBI))
                {
                    conn.Open();
                    using (SqlCommand cmd = new SqlCommand(query, conn))
                    {
                        using (SqlDataReader rdr = cmd.ExecuteReader())
                        {
                            while (rdr.Read())
                            {
                                string jobName = rdr["JobName"].ToString().Trim();
                                string status = rdr["LastStatus"].ToString();
                                biResults[jobName] = status;
                            }
                        }
                    }
                }
            }
            catch (Exception ex)
            {
                string cleanError = ex.Message.Replace("'", "").Replace("\"", "");
                biResults["ERROR_FLAG"] = "<b>שגיאת התחברות ל-home: " + cleanError + "</b>";
                Dts.Events.FireWarning(0, "BI Jobs Status Error", ex.Message, "", 0);
            }
            return biResults;
        }

        private string GetVal(Dictionary<int, string> d, int k)
        {
            if (d != null && d.ContainsKey(k)) return d[k];
            return "<b>טרם הסתיים</b>";
        }

        private string GetBIVal(Dictionary<string, string> d, string key)
        {
            if (d == null) return "<b>טרם עודכן</b>";
            if (d.ContainsKey("ERROR_FLAG")) return d["ERROR_FLAG"];

            string searchKey = key.Trim().ToLower();
            foreach (var kvp in d)
            {
                if (kvp.Key.ToLower().Contains(searchKey))
                {
                    return kvp.Value;
                }
            }
            return "<b>טרם עודכן</b>";
        }

        }

        private string RunCheckStatusOnly(string sourceFolder, string archiveFolder)
        {
            try
            {
                if (!Directory.Exists(sourceFolder)) return "FAILED";

                string[] sourceFiles = Directory.GetFiles(sourceFolder);
                if (sourceFiles.Length == 0) return "SUCCESS";

                if (!Directory.Exists(archiveFolder)) return "FAILED";

                var archiveFileNames = new HashSet<string>(
                    Directory.GetFiles(archiveFolder).Select(Path.GetFileName),
                    StringComparer.OrdinalIgnoreCase
                );

                foreach (string f in sourceFiles)
                {
                    string fileName = Path.GetFileNameWithoutExtension(f);
                    if (!fileName.Contains("_")) continue;

                    string id = fileName.Split('_').Last();

                    if (!archiveFileNames.Any(x => x.IndexOf(id, StringComparison.OrdinalIgnoreCase) >= 0))
                        return "FAILED";
                }
                return "SUCCESS";
            }
            catch { return "FAILED"; }
        }
