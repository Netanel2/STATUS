    // מנגנון העתקה ללוח עם חיווי ויזואלי + נפילה חכמה ל-execCommand
    // (נדרש כש-navigator.clipboard חסום ב-iframe / file:// / הקשר לא מאובטח)
    function copyText(elementId, btnElement) {
        const target = document.getElementById(elementId);
        if (!target) return;

        const textToCopy = target.tagName === 'TEXTAREA' ? target.value : target.innerText;
        if (!textToCopy || textToCopy.trim() === "") return;

        const showResult = (ok) => {
            const originalText = btnElement.innerText;
            btnElement.innerText = ok ? "הועתק!" : "נסה שוב";
            btnElement.classList.toggle('copied', ok);
            if (ok && typeof window.showCopyToast === 'function') {
                window.showCopyToast('הועתק ללוח');
            }
            setTimeout(() => {
                btnElement.innerText = originalText;
                btnElement.classList.remove('copied');
            }, 2000);
        };

        // שיטת גיבוי: textarea זמני + execCommand('copy') — עובד גם כשה-Clipboard API חסום
        const fallbackCopy = (txt) => {
            const ta = document.createElement('textarea');
            ta.value = txt;
            ta.setAttribute('readonly', '');
            ta.style.position = 'fixed';
            ta.style.top = '-9999px';
            ta.style.left = '-9999px';
            ta.style.opacity = '0';
            document.body.appendChild(ta);
            const sel = document.getSelection();
            const saved = (sel && sel.rangeCount > 0) ? sel.getRangeAt(0) : null;
            ta.focus();
            ta.select();
            try { ta.setSelectionRange(0, ta.value.length); } catch (e) {}
            let ok = false;
            try { ok = document.execCommand('copy'); } catch (e) { ok = false; }
            document.body.removeChild(ta);
            if (saved && sel) { sel.removeAllRanges(); sel.addRange(saved); }
            return ok;
        };

        if (navigator.clipboard && navigator.clipboard.writeText) {
            navigator.clipboard.writeText(textToCopy)
                .then(() => showResult(true))
                .catch(() => showResult(fallbackCopy(textToCopy)));
        } else {
            showResult(fallbackCopy(textToCopy));
        }
    }
	    // מנוע צביעת קוד אוטומטי ללא ספריות חיצוניות (Local Syntax Highlighter)
    function highlightCode(text, type = 'auto') {
        if (!text) return '';
        
        // מניעת הזרקות קוד והמרת תגיות לתצוגה בטוחה
        let html = text
            .replace(/&/g, "&amp;")
            .replace(/</g, "&lt;")
            .replace(/>/g, "&gt;");

        // אם זה קוד XML / HTML
        if (type === 'xml' || (type === 'auto' && text.trim().startsWith('<'))) {
            // צובע כל תג בנפרד באמצעות פונקציית-החלפה, ובונה את התוצאה ע"י שרשור.
            // קריטי: ה-regex רץ רק על הטקסט המקורי (html / inner) ולעולם לא על ה-spans
            // שכבר הזרקנו — כך נמנעת ההתנגשות שגרמה ל-class="hl-tag" לחזור כטקסט.
            return html.replace(/(&lt;)(\/?)([\s\S]*?)(&gt;)/g, function (match, lt, slash, inner, gt) {
                const attrRe = /([a-zA-Z_:][\w:.\-]*)\s*=\s*("[^"]*"|'[^']*')/g;
                let buf = '';
                let last = 0;
                let m;
                while ((m = attrRe.exec(inner)) !== null) {
                    // שם התג / רווחים שלפני השם-ערך — בצבע תג
                    if (m.index > last) {
                        buf += '<span class="hl-tag">' + inner.slice(last, m.index) + '</span>';
                    }
                    buf += '<span class="hl-attr">' + m[1] + '</span>='
                         + '<span class="hl-string">' + m[2] + '</span>';
                    last = m.index + m[0].length;
                }
                if (last < inner.length) {
                    buf += '<span class="hl-tag">' + inner.slice(last) + '</span>';
                }
                return '<span class="hl-tag">' + lt + slash + '</span>' + buf
                     + '<span class="hl-tag">' + gt + '</span>';
            });
        }

        // אם זה קוד SQL (הכלים: Beautify, בונה השאילתות, XML ל-SQL, SSIS)
        html = html
            .replace(/(--.*)/g, '<span class="hl-comment">$1</span>') // צביעת הערות UPDATE במקפים
            .replace(/'([^']*)'/g, '<span class="hl-string">\'$1\'</span>') // צביעת ערכים בגרשים
            .replace(/\b(1|0)\b/g, '<span class="hl-number">$1</span>'); // מספרים בסיסיים

        const sqlKeywords = [
            'SELECT', 'FROM', 'WHERE', 'UPDATE', 'SET', 'INSERT INTO', 'VALUES', 'CREATE TABLE',
            'LEFT JOIN', 'INNER JOIN', 'RIGHT JOIN', 'JOIN', 'GROUP BY', 'ORDER BY', 'WITH', 'NOLOCK',
            'AND', 'OR', 'IN', 'ON', 'TOP', 'CASE', 'WHEN', 'THEN', 'ELSE', 'END', 'IF', 'BEGIN', 
            'DECLARE', 'EXEC', 'IS', 'NULL', 'NOT', 'DROP', 'TABLE', 'GO'
        ];

        sqlKeywords.forEach(keyword => {
            const regex = new RegExp(`\\b${keyword}\\b`, 'g');
            html = html.replace(regex, `<span class="hl-keyword">${keyword}</span>`);
        });

        return html;
    }

        target.innerHTML = highlightCode(text, type);
    }


    // לוגיקת כלי 1: IN Builder

    function generateInBuilder() {
        const input = document.getElementById('inInput').value;
        const wrapQuotes = document.getElementById('wrapQuotes').checked;
        const wrapParentheses = document.getElementById('wrapParentheses').checked;
        const padZeros = document.getElementById('padZeros').checked; // קריאת הצ'קבוקס החדש
        const outputDiv = document.getElementById('inOutput');

        let lines = input.split('\n')
            .map(line => line.trim())
            .filter(line => line.length > 0);

        if (lines.length === 0) {
            outputDiv.innerText = '';
            return;
        }

        // אם המשתמש סימן את הצ'קבוקס, נשלים אפסים מובילים לכל שורה שהיא מספר נקי וקצרה מ-9
        if (padZeros) {
            lines = lines.map(line => {
                // בדיקה אם השורה מכילה רק ספרות
                if (/^\d+$/.test(line)) {
                    return line.padStart(9, '0');
                }
                return line; // אם זה מכיל אותיות, נשאיר כמו שזה
            });
        }

        const formatted = lines.map(line => wrapQuotes ? `'${line}'` : line);
        let result = formatted.join(',\n');

        if (wrapParentheses) {
            result = `(\n${result}\n)`;
        }

        outputDiv.innerText = result;
    }

    // לוגיקת כלי 2: Beautify (מנוע ייפור וסידור)
    function processBeautify() {
        const input = document.getElementById('beautifyInput').value.trim();
        const errorDiv = document.getElementById('beautifyError');
        const outputDiv = document.getElementById('beautifyOutput');

        errorDiv.style.display = 'none';

        let detectedType = currentBeautifyType;

        }
    }

    function detectCodeType(str) {
        if ((str.startsWith('{') && str.endsWith('}')) || (str.startsWith('[') && str.endsWith(']'))) {
            return 'json';
        }
        if (str.startsWith('<') && str.endsWith('>')) {
            return 'xml';
        }
        return 'sql';
    }

    function formatJSON(str) {
        const parsed = JSON.parse(str);
        return JSON.stringify(parsed, null, 4);
    }

        function formatXML(str) {
        // 1. ניקוי רווחים מיותרים ויצירת מערך נקי של תגים ותוכן
        let cleanXml = str.replace(/>\s*</g, '><').trim();
        let reg = /(<[^>]+>)([^<]*)/g;
        let matches = [...cleanXml.matchAll(reg)];
        
        let formatted = '';
        let indentLevel = 0;
        const tab = '    '; // 4 רווחים לכל רמה

        for (let i = 0; i < matches.length; i++) {
            let tag = matches[i][1];
            let text = matches[i][2] ? matches[i][2].trim() : '';

            // א. אם זה תג סגירה רגיל (למשל </AUTOMATED_INCIDENT>)
            if (tag.startsWith('</')) {
                indentLevel = Math.max(0, indentLevel - 1);
                formatted += (formatted ? '\n' : '') + tab.repeat(indentLevel) + tag;
            }
            // ב. אם זה תג בעיטה עצמי (למשל <ATTACHMENT /> או תג הערה <?xml ?>)
            else if (tag.endsWith('/>') || tag.startsWith('<?') || tag.startsWith('<!--')) {
                formatted += (formatted ? '\n' : '') + tab.repeat(indentLevel) + tag;
            }
            // ג. אם זה תג פתיחה רגיל (למשל <FORM> או <ATTACHMENTS>)
            else {
                // בדיקה אם תג הסגירה המתאים נמצא מיד בצמוד אליו או שהטקסט קצר (סגירה באותה שורה)
                let nextMatch = matches[i + 1];
                let isClosedNext = nextMatch && nextMatch[1] === `</${tag.substring(1, tag.indexOf('>')).split(' ')[0]}>`;

                if (isClosedNext) {
                    // מיישר את הפתיחה, התוכן והסגירה בשורה אחת מושלמת
                    formatted += (formatted ? '\n' : '') + tab.repeat(indentLevel) + tag + text + nextMatch[1];
                    i++; // מדלג על תג הסגירה שכבר טיפלנו בו
                } else {
                    // תג פתיחה מורכב (פותח בלוק חדש)
                    formatted += (formatted ? '\n' : '') + tab.repeat(indentLevel) + tag;
                    if (text) {
                        formatted += '\n' + tab.repeat(indentLevel + 1) + text;
                    }
                    indentLevel++;
                }
            }
        }

        if (formatted === '') {
            throw new Error("מבנה ה-XML אינו תקין או חסר סגירות.");
        }
        return formatted;
    }


    function formatSQL(str) {
        const keywords = [
            'SELECT', 'FROM', 'WHERE', 'LEFT JOIN', 'INNER JOIN', 
            'RIGHT JOIN', 'JOIN', 'GROUP BY', 'ORDER BY', 
            'AND', 'OR', 'HAVING', 'LIMIT'
        ];

        let cleanSql = str.replace(/\s+/g, ' ').trim();

        keywords.forEach(keyword => {
            const regex = new RegExp(`\\b${keyword}\\b`, 'gi');
            cleanSql = cleanSql.replace(regex, `\n${keyword}`);
        });

        cleanSql = cleanSql
            .replace(/,\s*/g, ',\n       ')
            .replace(/\s*=\s*/g, ' = ')
            .replace(/\s*<>\s*/g, ' <> ')
            .replace(/\s*>\s*/g, ' > ')
            .replace(/\s*<\s*/g, ' < ');

        return cleanSql.split('\n')
            .map(line => line.trimEnd())
            .filter(line => line.trim().length > 0)
            .join('\n');
    }

    function processDiff() {
        const original = document.getElementById('diffOriginal').value;
        const modified = document.getElementById('diffModified').value;
        const outputDiv = document.getElementById('diffOutput');

        if (!original && !modified) {
            outputDiv.innerHTML = "אנא הזן טקסט בשדות הקלט לצורך ביצוע השוואה.";
            return;
        }

        const tokenize = (str) => str.split(/(\s+)/);
        const outWords = tokenize(original);
        const editWords = tokenize(modified);

        const m = outWords.length;
        const n = editWords.length;
        const dp = Array.from({ length: m + 1 }, () => Array(n + 1).fill(0));

        for (let i = 1; i <= m; i++) {
            for (let j = 1; j <= n; j++) {
                if (outWords[i - 1] === editWords[j - 1]) {
                    dp[i][j] = dp[i - 1][j - 1] + 1;
                } else {
                    dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]);
                }
            }
        }

        let i = m, j = n;
        let htmlResult = [];

        while (i > 0 || j > 0) {
            if (i > 0 && j > 0 && outWords[i - 1] === editWords[j - 1]) {
                htmlResult.push(escapeHtml(outWords[i - 1]));
                i--; j--;
            } else if (j > 0 && (i === 0 || dp[i][j - 1] >= dp[i - 1][j])) {
                htmlResult.push(`<span class="diff-added">${escapeHtml(editWords[j - 1])}</span>`);
                j--;
            } else if (i > 0 && (j === 0 || dp[i][j - 1] < dp[i - 1][j])) {
                htmlResult.push(`<span class="diff-removed">${escapeHtml(outWords[i - 1])}</span>`);
                i--;
            }
        }

        outputDiv.innerHTML = htmlResult.reverse().join('');
    }

    function escapeHtml(text) {
        return text
            .replace(/&/g, "&amp;")
            .replace(/</g, "&lt;")
            .replace(/>/g, "&gt;")
            .replace(/"/g, "&quot;")
            .replace(/'/g, "&#39;");
    }

    function processXmlToSql() {

    function quoteColumn(name, dialect) {
        if (dialect === 'tsql') return `[${name}]`;
        if (dialect === 'mysql') return `\`${name}\``;
        return name;
    }

    function formatValue(val, nullEmpty) {
        if (!val || val.trim() === "") {
            return nullEmpty ? "NULL" : "''";
        }
        const escaped = val.replace(/'/g, "''");
        return `'${escaped}'`;
    }

      function convertFetchXmlToSql(xmlDoc, dialect) {
        const entityNodes = xmlDoc.getElementsByTagName("entity");
        let tableName = "unknown_table";
        let contextNode = xmlDoc.getElementsByTagName("fetch");

        if (entityNodes.length > 0) {
            const entityNode = entityNodes[0];
            tableName = entityNode.getAttribute("name") || "unknown_table";
            contextNode = entityNode;
        }
        
        const attrNodes = contextNode.getElementsByTagName("attribute");
        let columns = [];
        for (let i = 0; i < attrNodes.length; i++) {
            const colName = attrNodes[i].getAttribute("name");
            if (colName) columns.push(quoteColumn(colName, dialect));
        }
        if (columns.length === 0) columns.push("*");

        let sql = `SELECT\n    ${columns.join(',\n    ')}\nFROM ${tableName} WITH (NOLOCK)`;

        // פונקציית עזר רקורסיבית לסריקת פילטרים ותנאים כולל תמיכה ב-AND ו-OR
        function parseFilterNode(filterNode) {
            const filterType = (filterNode.getAttribute("type") || "and").toUpperCase(); // AND או OR
            let clauses = [];

            // 1. עיבוד תנאים ישירים בתוך הפילטר הנוכחי
            const conditionNodes = filterNode.children;
            for (let i = 0; i < conditionNodes.length; i++) {
                const node = conditionNodes[i];
                
                if (node.tagName.toLowerCase() === 'condition') {
                    const attr = node.getAttribute("attribute");
                    let op = (node.getAttribute("operator") || "eq").toLowerCase();
                    const val = node.getAttribute("value");
                    
                    if (attr) {
                        let sqlOp = "=";
                        if (op === 'ne') sqlOp = "<>";
                        if (op === 'gt') sqlOp = ">";
                        if (op === 'lt') sqlOp = "<";
                        if (op === 'like') sqlOp = "LIKE"; // תיקון באג ה-LIKE

                        const quotedAttr = quoteColumn(attr, dialect);
                        const formattedVal = val ? `'${val.replace(/'/g, "''")}'` : "NULL";
                        clauses.push(`${quotedAttr} ${sqlOp} ${formattedVal}`);
                    }
                }
                // 2. תמיכה בפילטרים מקוננים (רקורסיה בשביל בלוקים של OR פנימי)
                else if (node.tagName.toLowerCase() === 'filter') {
                    const subFilterResult = parseFilterNode(node);
                    if (subFilterResult) {
                        clauses.push(subFilterResult);
                    }
                }
            }

            if (clauses.length === 0) return "";
            if (clauses.length === 1) return clauses[0];
            
            // חיבור התנאים לפי סוג הפילטר ועטיפה בסוגריים לשמירה על הלוגיקה
            return `(${clauses.join(` ${filterType} `)})`;
        }

        // איתור הפילטר הראשי של הישות
        const rootFilterNodes = contextNode.children;
        let mainFilterResult = "";
        for (let i = 0; i < rootFilterNodes.length; i++) {
            if (rootFilterNodes[i].tagName.toLowerCase() === 'filter') {
                mainFilterResult = parseFilterNode(rootFilterNodes[i]);
                break;
            }
        }

        if (mainFilterResult) {
            // הסרת הסוגריים החיצוניים ביותר בשביל היופי של השאילתה
            if (mainFilterResult.startsWith('(') && mainFilterResult.endsWith(')')) {
                mainFilterResult = mainFilterResult.substring(1, mainFilterResult.length - 1);
            }
            sql += `\nWHERE \n    ${mainFilterResult};`;
        } else {
            sql += ";";
        }

        return sql;
    }


    function convertDataXmlToSql(xmlDoc, dialect, action, nullEmpty) {
        const root = xmlDoc.documentElement;
        let targetRecord = root;
        if (root.children.length > 0 && root.tagName.toLowerCase() !== 'row') {
            targetRecord = root.children[0];
        }

        const tableName = targetRecord.tagName;
        const fields = targetRecord.children;

        let colNames = [];
        let sampleValues = [];

        if (fields.length === 0) {
            colNames.push(targetRecord.tagName);
            sampleValues.push(targetRecord.textContent);
        } else {
            for (let i = 0; i < fields.length; i++) {
                colNames.push(fields[i].tagName);
                sampleValues.push(fields[i].textContent);
            }
        }

        let chosenAction = action;
        if (action === 'auto') chosenAction = 'insert';

        if (chosenAction === 'create') {
            let defs = colNames.map(name => `    ${quoteColumn(name, dialect)} VARCHAR(255)`);
            return `CREATE TABLE ${tableName} (\n${defs.join(',\n')}\n);`;
        }

        if (chosenAction === 'select') {
            let quotedCols = colNames.map(name => quoteColumn(name, dialect));
            return `SELECT\n    ${quotedCols.join(',\n    ')}\nFROM ${tableName};`;
        }

        let quotedCols = colNames.map(name => quoteColumn(name, dialect));
        let valStrings = sampleValues.map(v => formatValue(v, nullEmpty));

        return `INSERT INTO ${tableName} (\n    ${quotedCols.join(',\n    ')}\n)\nVALUES (\n    ${valStrings.join(',\n    ')}\n);`;
    }

        // פונקציית הליבה שמייצרת ומפצלת לשאילתות, ומנטרלת את ה-UPDATE שורה אחר שורה
    function buildFinalQuery() {
        if (!currentSelectedTableKey) return;

        const tableData = DB_SCHEMA[currentSelectedTableKey];
        const tableName = tableData.tableName;

        // --- 1. בניית רכיבי ה-SELECT (עמודות שנבחרו לשליפה) ---
        const selectCols = [];
        document.querySelectorAll(".builder-select-col:checked").forEach(cb => {
            selectCols.push(cb.value);
        });
        let selectFieldsText = selectCols.length > 0 ? "\n    " + selectCols.join(",\n    ") : " *";

        // --- 2. בניית תנאי ה-WHERE (תנאי חיפוש - משותף ל-SELECT ול-UPDATE) ---
        const whereClauses = [];
        document.querySelectorAll(".builder-where-col:checked").forEach(cb => {
            const colName = cb.value;
            // קריאת הערך אך ורק משדה קלט החיפוש (WHERE)
            const inputEl = document.querySelector(`.builder-where-input[data-column="${colName}"]`);
            const val = inputEl ? inputEl.value.trim() : "";
            const escapedVal = val.replace(/'/g, "''");
            whereClauses.push(`${colName} = '${escapedVal}'`);
        });

        // --- 3. בניית ערכי העדכון המפורשים של ה-SET (עבור ה-UPDATE בלבד) ---
        const setClauses = [];
        document.querySelectorAll(".builder-set-col:checked").forEach(cb => {
            const colName = cb.value;
            // קריאת הערך אך ורק משדה קלט העדכון החדש (SET)
            const inputEl = document.querySelector(`.builder-set-input[data-column="${colName}"]`);
            const val = inputEl ? inputEl.value.trim() : "";
            const escapedVal = val.replace(/'/g, "''");
            setClauses.push(`${colName} = '${escapedVal}'`);
        });

        // --- 4. הרכבה והצגה של שאילתת ה-SELECT המשודרגת עם WITH (NOLOCK) ---
        let selectQuery = `SELECT ${selectFieldsText}\nFROM \n    ${tableName} WITH (NOLOCK)`;
        if (whereClauses.length > 0) {
            selectQuery += `\nWHERE \n    ${whereClauses.join("\n    AND ")}`;
        }
        selectQuery += ";";
        
        if (document.getElementById("builderSelectOutputCode")) {
            document.getElementById("builderSelectOutputCode").innerText = selectQuery;
        }

        // --- 5. הרכבה והצגה של שאילתת ה-UPDATE המאובטחת (הזרקת -- לכל שורה בנפרד) ---
        let rawUpdateQuery = `UPDATE TOP (1) ${tableName}\nSET `;
        if (setClauses.length > 0) {
            rawUpdateQuery += "\n    " + setClauses.join(",\n    ");
        } else {
            rawUpdateQuery += "\n    [סמן שדות בעמודת SET והזן ערכים כדי להגדיר מה לעדכן]";
        }
        
        if (whereClauses.length > 0) {
            rawUpdateQuery += `\nWHERE \n    ${whereClauses.join("\n    AND ")}`;
        } else {
            rawUpdateQuery += `\nWHERE \n    [חובה לסמן שדות ב-WHERE ולהזין ערך למניעת עדכון גורף של כל הטבלה!]`;
        }
        rawUpdateQuery += ";";

        // סריקה שורה אחר שורה והוספת -- בראש כל שורה בנפרד באופן קשיח ואוטומטי
        let commentedUpdateQuery = rawUpdateQuery.split('\n').map(line => {
            let trimmed = line.trim();
            if (trimmed === '') return line;
            if (trimmed.startsWith('--')) return line;
            return '-- ' + line;
        }).join('\n');
        
        if (document.getElementById("builderUpdateOutputCode")) {
            document.getElementById("builderUpdateOutputCode").innerText = commentedUpdateQuery;
        }
    }

    // לוגיקת כלי 7: Unicode ⇄ מלל
    // ==========================================
    let currentUniMode = 'decode';

    function setUniMode(mode, element) {
        currentUniMode = mode;
        const parent = element.parentElement;
        parent.querySelectorAll('.segment-btn').forEach(btn => btn.classList.remove('active'));
        element.classList.add('active');
    }

    // פענוח כלל פורמטי הבריחה הנפוצים למלל קריא
    function decodeAllEscapes(str) {
        let s = str;

        // \u{XXXXX}  (ES6, כולל תווים מעל BMP)
        s = s.replace(/\\u\{([0-9a-fA-F]+)\}/g, (_, h) => String.fromCodePoint(parseInt(h, 16)));
        // \uXXXX  (זוגות surrogate מתחברים אוטומטית כשהם צמודים)
        s = s.replace(/\\u([0-9a-fA-F]{4})/g, (_, h) => String.fromCharCode(parseInt(h, 16)));
        // \xXX
        s = s.replace(/\\x([0-9a-fA-F]{2})/g, (_, h) => String.fromCharCode(parseInt(h, 16)));
        // ישויות HTML: &#xHHHH;  ו-  &#NNNN;
        s = s.replace(/&#x([0-9a-fA-F]+);/g, (_, h) => String.fromCodePoint(parseInt(h, 16)));
        s = s.replace(/&#(\d+);/g, (_, d) => String.fromCodePoint(parseInt(d, 10)));
        // קידוד URL (UTF-8 percent-encoding) — רק אם קיימים אחוזים תקפים
        if (/%[0-9a-fA-F]{2}/.test(s)) {
            try { s = decodeURIComponent(s); } catch (e) { /* משאירים כפי שהוא אם לא תקין */ }
        }
        // בריחות JSON בסיסיות
        s = s.replace(/\\n/g, '\n')
             .replace(/\\t/g, '\t')
             .replace(/\\r/g, '\r')
             .replace(/\\"/g, '"')
             .replace(/\\\//g, '/');

        return s;
    }

    // קידוד: כל תו לא-ASCII הופך ל-\uXXXX (ASCII נשאר קריא)
    function encodeToUnicode(str) {
        let out = '';
        for (const ch of str) {              // איטרציה לפי code point (תומך באמוג'י)
            const cp = ch.codePointAt(0);
            if (cp > 0xFFFF) {
                const high = Math.floor((cp - 0x10000) / 0x400) + 0xD800;
                const low  = ((cp - 0x10000) % 0x400) + 0xDC00;
                out += '\\u' + high.toString(16).padStart(4, '0')
                     + '\\u' + low.toString(16).padStart(4, '0');
            } else if (cp > 0x7F) {
                out += '\\u' + cp.toString(16).padStart(4, '0');
            } else {
                out += ch;
            }
        }
        return out;
    }

    function processUnicode() {
        const input = document.getElementById('uniInput').value;
        const outputDiv = document.getElementById('uniOutput');
        const errorDiv = document.getElementById('uniError');

        errorDiv.style.display = 'none';
        outputDiv.innerText = '';

        if (!input) return;

        try {
            const result = currentUniMode === 'encode'
                ? encodeToUnicode(input)
                : decodeAllEscapes(input);

            outputDiv.innerText = result;

            // כיוון תצוגה אוטומטי — RTL לתוצאה עברית, אחרת LTR
            const isRtl = /[\u0590-\u05FF\u0600-\u06FF]/.test(result);
            outputDiv.style.direction = isRtl && currentUniMode === 'decode' ? 'rtl' : 'ltr';
            outputDiv.style.textAlign = isRtl && currentUniMode === 'decode' ? 'right' : 'left';
        } catch (e) {
            errorDiv.innerText = 'שגיאה בהמרה: ' + e.message;
            errorDiv.style.display = 'block';
        }
    }
